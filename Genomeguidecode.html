const DISEASES = ['T2D','CVD','Cancer','Alzheimers','HTN'];

const DISEASE_LABELS = {

T2D: 'Type 2 Diabetes', CVD: 'Cardiovascular', Cancer: 'Cancer',

Alzheimers: "Alzheimer's", HTN: 'Hypertension'

};

function calcPRS(disease) {

let logSum = 0;

Object.values(state.snps).forEach(snp => {

if (snp.disease === disease) logSum += Math.log(snp.or);

});

// Baseline population average ≈ 0.3 (some always carry minor alleles)

const baseline = 0.3;

return baseline + logSum;

}

function calcLifestyleRisk() {

let score = 0;

// Exercise (max 10)

const ex = { '5+':0, '3-4':3, '1-2':7, '0':10 };

score += (ex[state.exercise] ?? 5);

// Diet (max 10)

const diet = { excellent:0, good:2, average:6, poor:10 };

score += (diet[state.diet] ?? 5);

// BMI (max 10)

const bmi = { normal:0, under:3, over:6, obese:10 };

score += (bmi[state.bmi] ?? 4);

// Sleep (max 6)

const sl = parseFloat(document.getElementById('sleep').value);

score += (sl >= 7 && sl <= 8) ? 0 : (sl >= 6 || sl <= 9) ? 3 : 6;

// Stress (max 6)

const st = parseInt(document.getElementById('stress').value);

score += st <= 3 ? 0 : st <= 6 ? 3 : 6;

// Smoking (max 8)

const sm = { never:0, former:3, current:8 };

score += (sm[state.smoking] ?? 0);

// Alcohol (max 8)

const al = { none:0, light:1, moderate:4, heavy:8 };

score += (al[state.alcohol] ?? 0);

// Normalize to 0–100

return Math.min(100, (score / 58) * 100);

}

function calcCombinedRisk(disease) {

const prs = calcPRS(disease);

const prsNorm = Math.min(100, (prs / 2.5) * 100);

const lifestyle = calcLifestyleRisk();

const famBonus = state.famHist.includes(disease) ? 20 : 0;

const combined = (prsNorm * 0.4) + (lifestyle * 0.4) + (famBonus * 0.2) + (Math.min(20,(state.age||40)/5)*0.2);

return Math.min(99, Math.round(combined));

}

function riskLevel(score) {

if (score < 35) return 'low';

if (score < 65) return 'moderate';

return 'high';

}

function riskEmoji(score) {

if (score < 35) return '';

if (score < 65) return '';

return '';

}

function fillClass(score) {

if (score < 35) return 'fill-low';

if (score < 65) return 'fill-mod';

return 'fill-high';

}

function formatAIText(text) {

return text

.replace(/\*\*(.*?)\*\*/g, '<strong>$1</strong>')

.replace(/\n\n/g, '<br><br>')

.replace(/\n- /g, '<br>• ')

.replace(/\n\d+\. /g, match => '<br>' + match.trim() + ' ');

}
